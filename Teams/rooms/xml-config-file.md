```XML
    </SendLogs>

        <ContentCameraId>USB\VID_046D&amp;PID_0843&amp;MI_00\7&amp;17446CF2&amp;0&amp;0000</ContentCameraId>

    </Devices>

        <ThemeName>Custom</ThemeName>

            <Audio default="false" enabled="false" />

        </Settings> 
    </CoordinatedMeetings>
</SkypeSettings>
```
